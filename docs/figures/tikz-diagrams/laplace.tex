\usepackage{mathtools}

\begin{document}

\definecolor{tabBlue}{HTML}{1f77b4}
\definecolor{tabOrange}{HTML}{ff7f0e}
\definecolor{tabGreen}{HTML}{2ca02c}
\definecolor{tabRed}{HTML}{d62728}
\definecolor{tabPurple}{HTML}{9467bd}
\definecolor{tabBrown}{HTML}{8c564b}
\definecolor{tabPink}{HTML}{e377c2}
\definecolor{tabGray}{HTML}{7f7f7f}
\definecolor{tabOlive}{HTML}{bcbd22}
\definecolor{tabCyan}{HTML}{17becf}

\begin{tikzpicture}[>={Stealth[inset=0pt,length=5pt]},
    thick,
    declare function={
        laplace(\x,\mu,\b)=1/(2*\b)*exp(-abs(\x-\mu)/\b);
        mu=0;
        b=0.5;
        % opacity
        o=0.1;
        xmax=2;
        ymax=1.2;
    }]
    
    % Shaded area under the entire curve
    \fill[black, opacity=o] plot[domain=-xmax:xmax,samples=150,smooth] 
    (\x,{laplace(\x,mu,b)}) -- (xmax,0) -| (-xmax,0) -- cycle;
    
    
    % Main distribution curve
    \draw[black] plot[domain=-xmax:xmax,samples=200,smooth] (\x,{laplace(\x,mu,b)});
    
    % Scale parameter
    \draw[tabOrange, densely dotted, <->, >={Bar[width=5, sep=-0.5]}] (-b,{laplace(-b,mu,b)}) -- (b,{laplace(b,mu,b)});
    
    % x-axis
    \draw[tabGray!80, -{>[sep=-2.5]}] (-xmax,0) -- (xmax,0);
    
    % Location parameter
    \draw[tabBlue, densely dotted, -{Rays[length=5, sep=-2.5]}] (0,{laplace(0,mu,b)-0.08}) -- (0,0);
\end{tikzpicture}
\end{document}
